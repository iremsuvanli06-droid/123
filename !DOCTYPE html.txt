let elixir = 0;
const elixirBar = document.getElementById('elixir-bar');
const elixirCount = document.getElementById('elixir-count');

// Her 1.5 saniyede 1 iksir artsın
setInterval(() => {
    if (elixir < 10) {
        elixir++;
        elixirCount.innerText = elixir;
        elixirBar.style.width = (elixir * 10) + "%";
    }
}, 1500);

function playCard(name, cost) {
    if (elixir >= cost) {
        elixir -= cost;
        alert(name + " sahaya sürüldü! Seni çok seviyor!");
    } else {
        alert("Yetersiz iksir! Biraz bekle.");
    }
}
